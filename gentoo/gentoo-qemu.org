* windows guest



** DONE qemu, blank image

cd /tmp

qemu-img create -f qcow2 WindowsVM.img 15G

qemu-system-x86_64 \
        -enable-kvm \
        -cpu host \
        -drive file=WindowsVM.img,if=virtio \
        -net nic -net user,hostname=windowsvm \
        -m 1G \
        -monitor stdio \
        -name Windows \
        "$@"
